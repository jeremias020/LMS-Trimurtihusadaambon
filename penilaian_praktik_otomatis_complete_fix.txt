Schema::create('practical_submissions', function (Blueprint $table) {
    $table->id();
    $table->foreignId('practical_id')->constrained()->onDelete('cascade');
    $table->foreignId('siswa_id')->constrained('users')->onDelete('cascade');
    $table->string('file_path')->nullable();
    $table->decimal('score', 5, 2)->nullable();
    $table->text('feedback')->nullable();
    $table->json('detail_penilaian')->nullable();
    $table->timestamp('submitted_at')->nullable();
    $table->timestamp('graded_at')->nullable();
    $table->foreignId('graded_by')->nullable()->constrained('users')->onDelete('set null');
    $table->timestamps();
});

// 2024_03_13_000002_create_nilai_praktik_table.php
Schema::create('nilai_praktik', function (Blueprint $table) {
    $table->id();
    $table->foreignId('submission_id')->constrained('practical_submissions')->onDelete('cascade');
    $table->foreignId('kriteria_id')->constrained()->onDelete('cascade');
    $table->decimal('nilai', 5, 2);
    $table->decimal('nilai_terbobot', 5, 2);
    $table->timestamps();
});
```

// Auto Assessment Form
public function autoAssessment(): View
{
    $students = User::where('role', 'siswa')->where('status', 'active')->with('kelas')->get();
    $practicals = Practical::where('guru_id', $guruId)->where('is_published', true)->with('subject')->get();
    return view('guru.penilaian.auto', compact('students', 'practicals'));
}

// Save Auto Assessment
public function saveAutoAssessment(Request $request): JsonResponse
{
    $submission = PracticalSubmission::updateOrCreate([
        'practical_id' => $request->practical_id,
        'siswa_id' => $request->student_id,
    ], [
        'score' => $request->score,
        'feedback' => $this->generateAutoFeedback($request->criteria),
        'graded_at' => now(),
        'graded_by' => $guruId,
        'detail_penilaian' => json_encode($request->criteria)
    ]);
    
    // Save individual criteria scores
    foreach ($request->criteria as $criteria) {
        \App\Models\NilaiPraktik::updateOrCreate([
            'submission_id' => $submission->id,
            'kriteria_id' => $criteria['id'],
        ], [
            'nilai' => $criteria['nilai'],
            'nilai_terbobot' => $criteria['nilai'] * 0.01
        ]);
    }
}
```
