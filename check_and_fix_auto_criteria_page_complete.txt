    $classes = Kelas::where('status', 'active')->get();
    $assignments = Assignment::where('guru_id', $guruId)->where('is_published', true)->get();
    $practicals = Practical::where('guru_id', $guruId)->where('is_published', true)->get();
    $students = User::where('role', 'siswa')->where('status', 'active')->get();
    $allCriteria = DB::table('criteria')->get();  // ✅ Added
    
    return view('guru.penilaian.auto_with_criteria', compact('subjects', 'classes', 'assignments', 'practicals', 'students', 'allCriteria'));
}
