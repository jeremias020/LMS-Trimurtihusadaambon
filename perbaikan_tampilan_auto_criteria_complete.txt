### **🔧 Enhanced Score Calculation**
```javascript
// Handle dynamic IDs and weights
function calculateScore() {
    let totalScore = 0;
    let totalWeight = 0;
    let checkedCount = 0;
    let totalCriteria = 0;

    Object.keys(assessmentCriteria).forEach(category => {
        assessmentCriteria[category].forEach((criterion, index) => {
            totalCriteria++;
            const checkboxId = criterion.id || 'criterion_' + index;
            const checkbox = document.getElementById(checkboxId);
            const valueSpan = document.getElementById(checkboxId + '_value');
            const weight = criterion.bobot || criterion.weight || 0.1;
            
            if (checkbox && checkbox.checked) {
                totalScore += 100 * weight;
                totalWeight += weight;
                checkedCount++;
                if (valueSpan) valueSpan.textContent = '100';
            } else if (valueSpan) {
                valueSpan.textContent = '0';
            }
        });
    });

    // Calculate final score (normalized by total weight)
    const finalScore = totalWeight > 0 ? (totalScore / totalWeight) : 0;
    
    // Update display with animation
    document.getElementById('totalScore').textContent = finalScore.toFixed(1);
    document.getElementById('scoreBar').style.width = finalScore + '%';
    
    // Update grade with color
    updateGradeDisplay(finalScore);
    
    // Generate comprehensive auto feedback
    generateAutoFeedback(finalScore, checkedCount, totalCriteria);
}
```

### **🔧 Comprehensive Auto Feedback**
```javascript
function generateAutoFeedback(score, checkedCount, totalCriteria) {
    const feedbackTextarea = document.getElementById('feedback');
    const completionRate = totalCriteria > 0 ? (checkedCount / totalCriteria) * 100 : 0;
    
    let feedback = '';
    
    if (score >= 90) {
        feedback = `🌟 SANGAT BAIK! (${score.toFixed(1)}/100)\n\n`;
        feedback += `Siswa menunjukkan performa yang luar biasa dengan ${checkedCount}/${totalCriteria} kriteria terpenuhi (${completionRate.toFixed(0)}%).\n\n`;
        feedback += `✅ Keunggulan:\n`;
        feedback += `- Semua aspek penilaian dikuasai dengan sangat baik\n`;
        feedback += `- Teknik pelaksanaan sesuai standar\n`;
        feedback += `- Sikap profesional sangat terpuji\n\n`;
        feedback += `💡 Rekomendasi: Pertahankan kualitas kerja yang excellent ini dan bisa menjadi contoh untuk siswa lain.`;
    } else if (score >= 80) {
        feedback = `👍 BAIK! (${score.toFixed(1)}/100)\n\n`;
        feedback += `Siswa menunjukkan performa yang baik dengan ${checkedCount}/${totalCriteria} kriteria terpenuhi (${completionRate.toFixed(0)}%).\n\n`;
        feedback += `✨ Keunggulan:\n`;
        feedback += `- Sebagian besar aspek penilaian terpenuhi\n`;
        feedback += `- Teknik pelaksanaan baik\n`;
        feedback += `- Keamanan pasien terjaga\n\n`;
        feedback += `🔧 Area untuk improvement:\n`;
        feedback += `- Perhatikan detail pada kriteria yang belum terpenuhi\n`;
        feedback += `- Tingkatkan konsistensi dalam dokumentasi\n\n`;
        feedback += `💡 Rekomendasi: Sudah sangat baik, tingkatkan sedikit lagi pada aspek yang belum sempurna.`;
    }
    // ... continue for all grade levels (C, D, E)
    
    feedbackTextarea.value = feedback;
}
```
