```javascript
function showNotification(message, type = 'info') {
    const notification = document.createElement('div');
    notification.className = `alert alert-${type} alert-dismissible fade show position-fixed`;
    notification.style.cssText = 'top: 20px; right: 20px; z-index: 9999; min-width: 300px;';
    notification.innerHTML = `

        <button type="button" class="btn-close" data-bs-dismiss="alert"></button>
    `;
    
    document.body.appendChild(notification);
    
    // Auto-remove after 5 seconds
    setTimeout(() => {
        if (notification.parentNode) {
            notification.remove();
        }
    }, 5000);
}
```

B. IMPROVED SCORE VALIDATION:
```javascript
function validateScore(score, maxScore) {
    if (score > maxScore) {
        showNotification(`Nilai tidak boleh melebihi ${maxScore}`, 'warning');
        return false;
    }
    if (score < 0) {
        showNotification('Nilai tidak boleh kurang dari 0', 'warning');
        return false;
    }
    return true;
}

// Real-time percentage calculation
function updatePercentageDisplay(currentScore, maxScore) {
    const percentageDisplay = document.getElementById('percentageDisplay');
    if (percentageDisplay && maxScore > 0) {
        const percentage = (currentScore / maxScore * 100).toFixed(1);
        percentageDisplay.textContent = `${percentage}% dari nilai maksimal`;
    }
}
```
